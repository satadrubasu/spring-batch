Create Procedure Insert_ToReadItem
 As
  DECLARE @i bigint 
  SET @i = 1
 WHILE(@i <= 10000000)
 BEGIN
  INSERT INTO ReadItem( ItemName,ItemNumber,ItemGroupId )
      Values ('BulkInsert',@i,@i)
   SET @i = @i + 1
 END
 Go

 Create Procedure LoadTestDataSP1000
 As
  DECLARE @i bigint 
  SET @i = 1
  WAITFOR DELAY '00:00:10';
 WHILE(@i <= 1000)
 BEGIN
  INSERT INTO ReadItem( ItemName,ItemNumber,ItemGroupId )
